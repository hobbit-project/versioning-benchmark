PREFIX foaf: <http://xmlns.com/foaf/0.1/> 
PREFIX cwork: <http://www.bbc.co.uk/ontologies/creativework/> 
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>

SELECT DISTINCT ?cwAboutOrMentions ?dbsb
FROM {{{graphVmax}}}
WHERE {
	 ?creativeWork ?cwAboutOrMentions ?v3 .
	 ?v3 foaf:homepage ?v2 .
	 ?v3 rdf:type ?dbsb
}
ORDER BY RAND()
LIMIT 1000
