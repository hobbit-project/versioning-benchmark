#  Query Type: 4, Historical single-version structured query

PREFIX foaf: <http://xmlns.com/foaf/0.1/> 
PREFIX cwork: <http://www.bbc.co.uk/ontologies/creativework/> 
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

SELECT DISTINCT ?cwAboutOrMentions ?cwAboutOrMentionsUri
FROM {{{graphVhistorical}}}
WHERE { 
	{ 
		?creativeWork ?cwAboutOrMentions ?cwAboutOrMentionsUri .
		?cwAboutOrMentionsUri rdfs:COMMENT ?v0 . 
		FILTER (lang (?v0) = 'en') 
	}
	UNION { ?cwAboutOrMentionsUri foaf:depiction ?v1 }
	UNION { ?cwAboutOrMentionsUri foaf:homepage ?v2 } 
}
LIMIT 1000
