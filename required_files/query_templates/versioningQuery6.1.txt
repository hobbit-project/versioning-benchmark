#  Query Type: 6, Single-delta structured queries (queries performed on changes of two consecutive versions (delta))
#  Query Description : 
#  Retreive the difference in the number of Blog Posts between two versions 

PREFIX cwork:<http://www.bbc.co.uk/ontologies/creativework/> 

SELECT ((?cnt_2 - ?cnt_1) AS ?blog_posts_diff) WHERE { 
{ 
   SELECT (COUNT(*) AS ?cnt_1) 
   WHERE { 
      GRAPH {{{graphVxMinusGraphVmax}}} { ?creativeWork a cwork:BlogPost } }
} . 
{ 
   SELECT (COUNT(*) AS ?cnt_2) 
   WHERE { 
      GRAPH {{{graphVxPlus1}}} { ?creativeWork a cwork:BlogPost } }
} }
